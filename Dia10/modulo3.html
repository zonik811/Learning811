let filtroActual = 'Todos';
let busquedaActual = '';

// üíæ GUARDAR EN LOCALSTORAGE
function guardarEnLocalStorage() {
  localStorage.setItem('pokemonData', JSON.stringify(pokemonDatabase));
}

// üìÇ CARGAR DEL LOCALSTORAGE
function cargarDelLocalStorage() {
  const datos = localStorage.getItem('pokemonData');
  if (datos) {
    pokemonDatabase = JSON.parse(datos);
  }
}

// üü† RENDERIZAR CON FILTROS
function renderizarPokemon() {
  const grid = document.getElementById('pokemonGrid');
  grid.innerHTML = '';
  
  const pokemonFiltrados = pokemonDatabase.filter(pokemon => {
    const coincideNombre = pokemon.nombre.toLowerCase().includes(busquedaActual.toLowerCase());
    const coincideTipo = filtroActual === 'Todos' || pokemon.tipo.includes(filtroActual);
    return coincideNombre && coincideTipo;
  });
  
  if (pokemonFiltrados.length === 0) {
    grid.innerHTML = '<p>No se encontraron Pok√©mon :/</p>';
    return;
  }
  
  pokemonFiltrados.forEach(pokemon => {
    const card = `
      <div class="pokemon-card">
        <h2>${pokemon.nombre}</h2>
        <p>${pokemon.tipo}</p>
        <p>HP: ${pokemon.hp} | Ataque: ${pokemon.ataque}</p>
        <img src="${pokemon.imagen}" alt="${pokemon.nombre}">
        <div class="pokemon-actions">
          <button class="action-btn favorite-btn ${pokemon.favorito ? 'active' : ''}" onclick="toggleFavorito(${pokemon.id})">‚≠ê Favorito</button>
          <button class="action-btn delete-btn" onclick="eliminarPokemon(${pokemon.id})">üóëÔ∏è Eliminar</button>
        </div>
      </div>
    `;
    grid.innerHTML += card;
  });
}

function setupSearchListener() {

function setupFilterListeners() {

  botones.forEach(boton => {
    boton.addEventListener('click', (e) => {
      botones.forEach(b => b.classList.remove('active'));
      e.target.classList.add('active');
      filtroActual = e.target.dataset.filter;
      renderizarPokemon();
    });
  });
}

// ‚ûï AGREGAR POK√âMON
function agregarPokemon() {
  const nombre = document.getElementById('pokemonName').value.trim();
  const tipo = document.getElementById('pokemonType').value.trim();
  const hp = parseInt(document.getElementById('pokemonHp').value);
  const ataque = parseInt(document.getElementById('pokemonAtaque').value);
  const imagen = document.getElementById('pokemonImagen').value.trim();

  if (!nombre || !tipo || !hp || !ataque || !imagen) {
    alert('Completa todos los campos');
    return;
  }

  const id = Math.max(...pokemonDatabase.map(p => p.id), 0) + 1;
  const nuevoPokemon = { id, nombre, tipo, hp, ataque, imagen, favorito: false };
  pokemonDatabase.push(nuevoPokemon);
  guardarEnLocalStorage();
  renderizarPokemon();

  document.getElementById('pokemonName').value = '';
  document.getElementById('pokemonType').value = '';
  document.getElementById('pokemonHp').value = '';
  document.getElementById('pokemonAtaque').value = '';
  document.getElementById('pokemonImagen').value = '';
}

// üóëÔ∏è ELIMINAR POK√âMON
function eliminarPokemon(id) {
  const indice = pokemonDatabase.findIndex(p => p.id === id);
  if (indice !== -1) {
    pokemonDatabase.splice(indice, 1);
    guardarEnLocalStorage();
    renderizarPokemon();
  }
}

// ‚≠ê MARCAR/DESMARCAR FAVORITO
function toggleFavorito(id) {
  const pokemon = pokemonDatabase.find(p => p.id === id);
  if (pokemon) {
    pokemon.favorito = !pokemon.favorito;
    guardarEnLocalStorage();
    renderizarPokemon();
  }
}

}
